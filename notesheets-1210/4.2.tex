  Let \(f\) be twice differentiable on an interval \((a,b)\). Then,
  \begin{enumerate}
  \item If \(f''(x) > 0\) for each value of \(x\) in \((a,b)\), then
  \item If \(f''(x) < 0\) for each value of \(x\) in \((a,b)\), then
  \end{enumerate}
\end{thrm}

\begin{thrm}
  Let \(f\) be a twice differentiable function. Then, if \(f'(c) = 0\)
  and
  \begin{enumerate}
  \item \(f''(c) < 0\), then
  \item \(f''(c) > 0\), then
  \item \(f''(c) = 0\), then
  \end{enumerate}
\end{thrm}

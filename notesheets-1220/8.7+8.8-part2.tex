\begin{thrm}
  If \(f(x,y)\) is integrable over the plane region \(R\), then its
  average value over \(R\) is given by \[
    \iint_R
  \]
\end{thrm}
\begin{ex}
  Find the average value of \(f(x,y) = 6x^2y\) over \(R = \{(x,y) \mid
  0 \leq x \leq 1; 0 \leq y \leq 3\}\). 
\end{ex}

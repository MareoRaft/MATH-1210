\begin{thrm}
  Properties of (facts about) continuous functions:
  \begin{enumerate}
    \item The constant function
    \item The identity function

    If $f$ and $g$ are continuous at $x=p$, then:
    \item $f(x)^n$
    \item $f \pm g$
    \item $fg$
    \item $f/g$
  \end{enumerate}
\end{thrm}
